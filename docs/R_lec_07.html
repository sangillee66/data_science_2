your_url <- "https://openapi.naver.com/v1/search/news.json"

  req_headers(
    "X-Naver-Client-ID" = your_ClientID,
    "X-Naver-Client-Secret" = your_Client_Secret
  ) |> 
  req_url_query(
    query = search,
    display = 100,
    start = 1,
    sort = "date"
  ) |> 
  req_perform() |> 
  resp_body_json() 
naver_data <- data.frame(t(sapply(result_naver$items, data.frame)))
